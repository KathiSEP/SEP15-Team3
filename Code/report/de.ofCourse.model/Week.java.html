package de.ofCourse.model;

/**
 * Contains each week day's content
 * 
 * @author Patrick Cretu
 *
 */
public class Week {

	/**
	 * The hour of the tuples start time
	 */
	private String hour;
	
	/**
	 * The week's Monday content
	 */
	private String monday;
	
	/**
	 * The week's Tuesday content
	 */
	private String tuesday;
	
	/**
	 * The week's Wednesday content
	 */
	private String wednesday;
	
	/**
	 * The week's Thursday content
	 */
	private String thursday;
	
	/**
	 * The week's Friday content
	 */
	private String friday;
	
	/**
	 * The week's Saturday content
	 */
	private String saturday;
	
	/**
	 * The week's Sunday content
	 */
	private String sunday;

	/**
	 * 
	 * @return the tuple's start time
	 */
	public String getHour() {
		return hour;
	}

	 * Sets the week's start time
	 * 
	 * @param hour
	 *           the start time
	 */
	public void setHour(String hour) {
		this.hour = hour;
	}

	 * 
	 * @return the week's Monday content

		return monday;
